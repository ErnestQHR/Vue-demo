# WordPress REST API 配置示例
# 将此文件重命名为 .env 后使用

# WordPress REST API 基础 URL
# 开发环境示例：http://localhost:8080/wp-json/wp/v2
# 生产环境示例：https://your-domain.com/wp-json/wp/v2
VITE_WP_API_URL=https://your-wordpress-site.com/wp-json/wp/v2

# 是否启用 WordPress 数据源（true/false）
# false 时会使用本地静态 JSON 数据
VITE_USE_WORDPRESS=false

# 应用基础路径（GitHub Pages 部署时需要）
VITE_BASE_URL=/Vue-demo/
